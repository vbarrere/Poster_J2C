        \vspace{0.3cm}

        \textbf{\underline{Core-shell index $\eta$}}

        \vspace{0.3cm}

        Quantifies the degree of segregation in nanoalloys:

        \vspace{0.2cm}

        \begin{equation}
            \eta = \textcolor{approachOne}{2\left| \frac{n^{out}_{Ag}}{n^{out}_{Ag} + n^{out}_{Co}} - \frac{n_{Ag}}{n_{tot}}\right| + 2\left| \frac{n^{in}_{Ag}}{n^{in}_{Ag} + n^{in}_{Co}} - \frac{n_{Ag}}{n_{tot}}\right|} - \textcolor{termThree}{\frac{d_{com}}{2r_{g}}}
        \end{equation}

        \vspace{0.3cm}

        \begin{center}
            \begin{tikzpicture}[x=\linewidth, y=1cm,
                ibox/.style={rounded corners=4pt, minimum height=1.2cm, text width=0.85\linewidth, align=left, inner sep=8pt}]

                \node[ibox, fill=approachOne!12] at (0.45, 4.5) {
                    \textbf{Terms 1 \& 2:} deviation of \textcolor{colorAg}{\textbf{Ag}} concentration in outer/inner regions from overall composition
                };
                \node[ibox, fill=termThree!12] at (0.45, 2.5) {
                    \textbf{Term 3:} distance between centers of mass of the two species, normalized by $r_g$
                };
                \node[ibox, fill=approachThree!12] at (0.45, 0.5) {
                    $\eta \uparrow$ $\rightarrow$ \textbf{core--shell} \quad\quad $\eta \downarrow$ $\rightarrow$ \textbf{mixed / Janus}
                };
            \end{tikzpicture}
