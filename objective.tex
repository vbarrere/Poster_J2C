\block{Objective}{
        
    \vspace{0.5cm}
    
    Determine the structure of nanoalloys from HRTEM images using deep learning.

    \vspace{1.5cm}
        
    \textbf{\underline{Steps:}}
    \vspace{1cm}
    \begin{itemize}
        \item Development of a \textbf{molecular dynamic (MD) simulation} ;
        \vspace{0.5cm}
        \item Creation of a \textbf{dataset of nanoalloys} using MD simulations ;
        \vspace{0.5cm}
        \item Generation of \textbf{HRTEM images} from the dataset ;
        \vspace{0.5cm}
        \item Use of \textbf{deep learning to analyze HRTEM images} and extract information about the structure and properties of nanoalloys.
    \end{itemize}
}
